server {
    server_name dnelectronicsshop.com www.dnelectronicsshop.com;

    location = /favicon.ico { access_log off; log_not_found off; }
    location /static/ {
        alias /opt/DNELECTRONICS/dnelectronics/static/;
    }

    location /media/ {
        alias /opt/DNELECTRONICS/dnelectronics/media/;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/opt/DNELECTRONICS/dnelectronics/dnelectronics.sock;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/dnelectronicsshop.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/dnelectronicsshop.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}


server {
    if ($host = www.dnelectronicsshop.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = dnelectronicsshop.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name dnelectronicsshop.com www.dnelectronicsshop.com;
    return 404; # managed by Certbot




}
